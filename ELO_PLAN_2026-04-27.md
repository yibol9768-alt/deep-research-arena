```python
# 输入: cited_url, quoted_span (≤200 chars)
# 输出: score ∈ {0, 1}, evidence = matched substring
def quote_match(cited_url, quoted_span, min_overlap=0.7) -> dict:
    page_text = fetch(cited_url, timeout=10).text  # via shim
    # rapidfuzz partial_ratio between quoted_span and any 250-char window
```
